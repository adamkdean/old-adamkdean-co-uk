    var flickr_sizes = {
        'square':       '_s',
        'large_square': '_q',
        'thumbnail':    '_t',
        'small':        '_m',
        'small_320':    '_n',
        'medium':       '',
        'medium_640':   '_z',
        'medium_800':   '_c',
        'large':        '_b',
        'large_1600':   '_h',
        'large_2048':   '_k',
        'original':     '_o'
    };
